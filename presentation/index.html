HTML 
								<div class="grid">
									<header>Header</header>
									<aside>Sidebar</aside>
									<main>Main</main>
									<footer>Footer</footer>
								</div>
CSS 
								.container {
									display: grid;
									grid-gap: 15px;
									grid-template-columns: 200px 1fr;
									grid-template-rows: 50px 70% 50px;
									grid-template-areas:
										"header header"
										"aside main"
										"footer footer";
								}
								header {
									grid-area: header;
									background-color: #66A5AD;
								}
								aside {
									grid-area: aside;
									background-color: #AEBD38;
								}
								main {
									grid-area: main;
									background-color: #CE5A57;
								}
								footer {
									grid-area: footer;
									background-color: #EBDCB2;
								}
